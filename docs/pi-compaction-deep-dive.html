一、概述：为什么需要 Compaction

LLM 有固定的 context window（如 200k tokens）。长对话中，历史消息 + system prompt + tool results 会快速消耗 token 预算。Compaction 的目标是：

  • 压缩旧消息为结构化摘要，释放 token 空间
  • 保留近期消息原文，确保模型有足够工作记忆
  • 累积追踪文件操作，让模型知道曾经读/写过哪些文件
  • 增量更新，多次 compaction 不重复摘要已摘要的内容
Compaction 前后对比
Before compaction: ┌──────────────────────────────────────────────────────────────────┐ │ LLM sees ALL messages: │ │ [system] [user₁] [ass₁] [tool₁] [user₂] [ass₂] [tool₂] ... │ │ ← 可能已经 150k tokens │ └──────────────────────────────────────────────────────────────────┘ After compaction: ┌──────────────────────────────────────────────────────────────────┐ │ LLM sees: │ │ [system] [summary (~2k tokens)] [user₅] [ass₅] [tool₅] ... │ │ ← 压缩到 ~25k tokens │ └──────────────────────────────────────────────────────────────────┘

  if (!settings.enabled) return false;

}

2.4 Token 估算策略

estimateContextTokens() 的双层策略

不是每次都从头估算所有消息，而是利用 LLM 返回的 usage 数据：

  1. 找到最后一条带 usage 的 assistant 消息（真实 token 计数）
  2. 只对 usage 之后的新消息用 chars/4 启发式估算
  3. 总量 = usageTokens + trailingTokens

如果没有任何 usage 数据（如首次对话），则对所有消息用 chars/4 估算。

// Token 估算 - chars/4 启发式（保守，会高估）
function estimateTokens(message) {
  let chars = 0;
  switch (message.role) {
    case "user":
      chars = estimateTextAndImageContentChars(message.content);
      // 图片统一按 4800 字符估算
      return Math.ceil(chars / 4);
    case "assistant":
      // text + thinking + toolCall args 全部累加
      for (const block of message.content) {
        if (block.type === "text") chars += block.text.length;
        if (block.type === "thinking") chars += block.thinking.length;
        if (block.type === "toolCall") chars += block.name.length + JSON.stringify(block.arguments).length;
      }
      return Math.ceil(chars / 4);
    case "toolResult":
      chars = estimateTextAndImageContentChars(message.content);
      return Math.ceil(chars / 4);
    // ...其他类型类似
  }
}

三、核心算法流程

Step 1: prepareCompaction()

确定压缩范围，不做 LLM 调用

Step 2: compact()

调 LLM 生成摘要

Step 3: SessionManager.save()

写入 CompactionEntry 到 JSONL

Step 4: Session Reload

重新构建上下文，LLM 看到 summary + 保留的消息

3.1 prepareCompaction() 详解

function prepareCompaction(pathEntries, settings) {
  // ❶ 如果最后一条就是 compaction，跳过（刚压缩过）
  if (pathEntries[pathEntries.length - 1].type === "compaction") return undefined;

  // ❷ 找到上次的 CompactionEntry
  let prevCompactionIndex = -1;
  for (let i = pathEntries.length - 1; i >= 0; i--) {
    if (pathEntries[i].type === "compaction") { prevCompactionIndex = i; break; }
  }

  // ❸ 确定 boundary
  //    boundaryStart = 上次 firstKeptEntryId 的位置（即上次保留的第一条消息）
  //    boundaryEnd   = 全部 entries 末尾
  let boundaryStart = 0;
  if (prevCompactionIndex >= 0) {
    previousSummary = prevCompaction.summary;  // 用于增量更新
    boundaryStart = findIndex(prevCompaction.firstKeptEntryId);
  }

  // ❹ 估算当前 context token 总量
  const tokensBefore = estimateContextTokens(buildSessionContext(pathEntries).messages).tokens;

  // ❺ 找切割点：保留最近 keepRecentTokens (20k) 的消息
  const cutPoint = findCutPoint(pathEntries, boundaryStart, boundaryEnd, settings.keepRecentTokens);

  // ❻ 提取要摘要的消息
  const messagesToSummarize = entries[boundaryStart..historyEnd].messages;
  const turnPrefixMessages  = entries[turnStart..cutPoint].messages;  // if split turn

  // ❼ 提取文件操作
  const fileOps = extractFileOperations(messagesToSummarize, entries, prevCompactionIndex);

  return { firstKeptEntryId, messagesToSummarize, turnPrefixMessages, isSplitTurn, ... };
}
关键细节：boundaryStart 从上次 firstKeptEntryId 的位置开始（而不是 CompactionEntry 本身的位置）。这意味着上次被保留的消息，如果这次超出预算，会被纳入本次摘要。确保不会丢失信息。

四、切割点选择算法

4.1 有效切割点

消息类型可以切割？原因
user✓用户消息是 turn 边界
assistant✓可以切在 turn 中间（split turn）
bashExecution✓等价于 user message
custom / branchSummary✓自定义消息可作为边界
toolResult✗必须跟在 toolCall 后面，切断会导致模型困惑

4.2 算法步骤

function findCutPoint(entries, startIndex, endIndex, keepRecentTokens) {
  // ❶ 收集所有有效切割点的索引
  const cutPoints = findValidCutPoints(entries, startIndex, endIndex);

  // ❷ 从最新消息向前走，累加 token
  let accumulatedTokens = 0;
  let cutIndex = cutPoints[0]; // 默认：保留所有（最保守）

  for (let i = endIndex - 1; i >= startIndex; i--) {
    if (entries[i].type !== "message") continue;
    accumulatedTokens += estimateTokens(entries[i].message);

    // ❸ 累计超过预算 → 在此处找最近的有效切点
    if (accumulatedTokens >= keepRecentTokens) {
      cutIndex = cutPoints.find(c => c >= i);
      break;
    }
  }

  // ❹ 向前吸收非消息 entries（bash settings、model change 等）
  while (cutIndex > startIndex) {
    if (entries[cutIndex - 1].type === "compaction") break;
    if (entries[cutIndex - 1].type === "message") break;
    cutIndex--; // 包含这个非消息 entry
  }

  // ❺ 判断是否 Split Turn
  const isUserMessage = entries[cutIndex].message?.role === "user";
  if (!isUserMessage) {
    turnStartIndex = 向前找这个 turn 的 user message;
    isSplitTurn = true;
  }

  return { firstKeptEntryIndex: cutIndex, turnStartIndex, isSplitTurn };
}
可视化切割过程
entries: [hdr] [u₁] [a₁] [t₁] [u₂] [a₂] [t₂] [t₃] [u₃] [a₃] [t₄] [u₄] [a₄] ↑ keepRecentTokens 预算用完 cutPoints: [u₁=1] [a₁=2] [u₂=4] [a₂=5] [u₃=8] [a₃=9] [u₄=11] [a₄=12] 选择 cutIndex = 8 (u₃) ← 第一个 >= i 的有效切点 结果： messagesToSummarize = [u₁, a₁, t₁, u₂, a₂, t₂, t₃] ← 送去摘要 kept messages = [u₃, a₃, t₄, u₄, a₄] ← 保留原文 isSplitTurn = false ← 切在 user 消息上，完整 turn 边界

5.2 serializeConversation()

把 LLM Messages 序列化为文本格式，防止模型把它当作对话来继续：

function serializeConversation(messages) {
  const parts = [];
  for (const msg of messages) {
    if (msg.role === "user") {
      parts.push(`[User]: ${textContent}`);
    }
    else if (msg.role === "assistant") {
      // thinking 单独标注
      if (thinkingParts.length > 0)
        parts.push(`[Assistant thinking]: ${thinkingParts.join("\n")}`);
      if (textParts.length > 0)
        parts.push(`[Assistant]: ${textParts.join("\n")}`);
      if (toolCalls.length > 0)
        parts.push(`[Assistant tool calls]: read(path="foo.ts"); edit(path="bar.ts", ...)`);
    }
    else if (msg.role === "toolResult") {
      // ⚠️ 截断到 2000 字符！
      parts.push(`[Tool result]: ${truncateForSummary(content, 2000)}`);
    }
  }
  return parts.join("\n\n");
}
Tool result 截断：这是 compaction 能节省大量 token 的关键。tool result 经常是文件内容（几千行代码）或 bash 输出，截断到 2000 字符后交给摘要模型。摘要模型只需知道"读了什么文件、结果大致是什么"，不需要完整内容。

6.1 System Prompt（固定）

Do NOT continue the conversation. Do NOT respond to any questions in the

</conversation>

Use this EXACT format:

[What is the user trying to accomplish?]

## Constraints & Preferences
- [Any constraints...]

## Progress

- [x] [Completed tasks]

## Key Decisions
- **[Decision]**: [Rationale]

## Next Steps
1. [Ordered list...]

## Critical Context
- [Data needed to continue]

Keep each section concise. Preserve exact file paths, function names, and error messages.

6.3 增量更新 Prompt

如果已有前次摘要（previousSummary），使用不同的 prompt：

<conversation>
[只有本次新增的消息]
</conversation>

<previous-summary>
[上次的摘要内容]
</previous-summary>

The messages above are NEW conversation messages to incorporate into the existing
summary provided in <previous-summary> tags.

Update the existing structured summary with new information. RULES:
- PRESERVE all existing information from the previous summary
- ADD new progress, decisions, and context
- UPDATE Progress section: move "In Progress" → "Done" when completed
- UPDATE "Next Steps" based on what was accomplished
- PRESERVE exact file paths, function names, and error messages
- If something is no longer relevant, you may remove it

[...same format...]
增量 vs 全量的意义：每次 compaction 只处理"上次保留的消息"到"本次切割点"之间的新消息，配合 previousSummary 做增量合并。这意味着第 N 次 compaction 不需要重新阅读前 N-1 次已经摘要过的内容，只看新增部分。

6.4 maxTokens 计算

七、Split Turn 处理

7.1 什么是 Split Turn

一个 turn = user message + 其后所有 assistant responses + tool calls，直到下一个 user message。

当单个 turn 的 token 量 > keepRecentTokens 时，不得不切在 turn 中间：

一个超长 turn： [user₁] [ass₁] [tool₁] [ass₂] [tool₂] [tool₃] [ass₃] [tool₄] [ass₄] ↑ ↑ turnStartIndex = 0 cutIndex = 6 (ass₃) isSplitTurn = true ← 因为 cutIndex 不是 user message turnPrefixMessages = [user₁, ass₁, tool₁, ass₂, tool₂, tool₃] ← turn 的前半部分 kept messages = [ass₃, tool₄, ass₄] ← 保留的后半部分

7.2 Split Turn 的摘要策略

async function compact(preparation, model, ...) {
  if (isSplitTurn && turnPrefixMessages.length > 0) {
    // 🔀 并行生成两份摘要
    const [historySummary, turnPrefixSummary] = await Promise.all([
      // 1. 历史摘要（之前的完整 turns）
      messagesToSummarize.length > 0
        ? generateSummary(messagesToSummarize, model, ...)
        : "No prior history.",
      // 2. Turn 前缀摘要（被切开的 turn 的前半部分）
      generateTurnPrefixSummary(turnPrefixMessages, model, ...),
    ]);

    // 合并
    summary = `${historySummary}\n\n---\n\n**Turn Context (split turn):**\n\n${turnPrefixSummary}`;
  } else {
    // 正常情况：只生成历史摘要
    summary = await generateSummary(messagesToSummarize, model, ...);
  }
}

7.3 Turn 前缀摘要的专用 Prompt

This is the PREFIX of a turn that was too large to keep.
The SUFFIX (recent work) is retained.

Summarize the prefix to provide context for the retained suffix:

## Original Request
[What did the user ask for in this turn?]

## Early Progress
- [Key decisions and work done in the prefix]

## Context for Suffix
- [Information needed to understand the retained recent work]

Be concise. Focus on what's needed to understand the kept suffix.
为什么要分开？Turn 前缀摘要用更简短的格式（只有 3 个 section），且 maxTokens 预算更小（0.5 * reserveTokens vs 历史摘要的 0.8 * reserveTokens）。因为它的目的只是给保留的后半部分提供上下文，不需要完整的项目状态记录。

八、文件追踪

8.1 数据结构

interface FileOperations {
  read: Set<string>;     // read 工具读取的文件
  written: Set<string>;  // write 工具写入的文件
  edited: Set<string>;   // edit 工具编辑的文件
}

8.2 提取逻辑

function extractFileOpsFromMessage(message, fileOps) {
  if (message.role !== "assistant") return;
  for (const block of message.content) {
    if (block.type !== "toolCall") continue;
    const path = block.arguments?.path;
    if (!path) continue;
    switch (block.name) {
      case "read":  fileOps.read.add(path);    break;
      case "write": fileOps.written.add(path); break;
      case "edit":  fileOps.edited.add(path);  break;
    }
  }
}

8.3 累积策略

每次 compaction 时，从两个来源收集文件操作：

  1. 前次 CompactionEntry.details：继承历史的 readFiles + modifiedFiles
  2. 本次要摘要的消息：提取新的 tool call 中的文件路径

8.4 输出格式

function computeFileLists(fileOps) {
  const modified = new Set([...fileOps.edited, ...fileOps.written]);
  const readOnly = [...fileOps.read].filter(f => !modified.has(f)).sort();
  const modifiedFiles = [...modified].sort();
  return { readFiles: readOnly, modifiedFiles };
}

// 附加到摘要末尾：
// <read-files>
// src/utils.ts
// src/types.ts
// </read-files>
//
// <modified-files>
// src/config.ts
// src/index.ts
// </modified-files>
注意：readFiles 只包含"只读未修改"的文件。如果一个文件先被 read 后被 edit，它只出现在 modifiedFiles 中。

十、分支摘要（Branch Summarization）

与 Compaction 使用相同的序列化和摘要格式，但触发时机和目的不同：

Compaction
  • 触发：context 即将溢出
  • 目的：释放 token 空间
  • 效果：旧消息被摘要替代
  • 累积：与前次摘要增量合并
Branch Summarization
  • 触发：/tree 切换到另一分支
  • 目的：保留被离开分支的上下文
  • 效果：摘要注入新分支开头
  • 策略：从 leaf 向上走到 common ancestor

10.1 Branch Summary 收集逻辑

function collectEntriesForBranchSummary(session, oldLeafId, targetId) {
  // 找共同祖先（两条路径上最深的共享节点）
  const oldPath = session.getBranch(oldLeafId);
  const targetPath = session.getBranch(targetId);
  const commonAncestorId = findDeepestCommon(oldPath, targetPath);

  // 从 old leaf 往回走到 common ancestor，收集 entries
  let entries = [];
  let current = oldLeafId;
  while (current !== commonAncestorId) {
    entries.push(session.getEntry(current));
    current = entry.parentId;
  }
  entries.reverse(); // 变为时间顺序
  return { entries, commonAncestorId };
}

10.2 Token Budget 控制

function prepareBranchEntries(entries, tokenBudget) {
  // 从最新向最旧走，添加消息直到超出预算
  // 优先保留最近的上下文
  // summary/compaction entries 即使超出 90% 预算也尝试保留（它们是重要上下文）
  for (let i = entries.length - 1; i >= 0; i--) {
    const tokens = estimateTokens(message);
    if (totalTokens + tokens > tokenBudget) {
      if (isSummaryEntry && totalTokens < tokenBudget * 0.9) {
        messages.unshift(message); // 重要，强行塞入
      }
      break;
    }
    messages.unshift(message);
    totalTokens += tokens;
  }
}

十二、配置参数一览

参数默认值说明调优建议
enabled true 是否启用自动 compaction 生产环境建议保持开启
reserveTokens 16384 预留给 LLM 回复的 token 空间 如果模型回复经常被截断可以调大
keepRecentTokens 20000 保留最近消息的 token 量 越大 = 保留越多原文 = compaction 越频繁；
太小 = 模型容易丢失近期上下文

在 Neezy Studio 中的配置方式

// pi-agent.ts
function buildSettingsManager(cwd, agentDir) {
  const sm = SettingsManager.create(cwd, agentDir);
  sm.applyOverrides({
    defaultThinkingLevel: level,
    compaction: { enabled: true },  // ← 这里
  });

当前 Neezy Studio 的状况：只设置了 enabled: true，reserveTokens 和 keepRecentTokens 使用 SDK 默认值。没有自定义 compaction 策略，没有利用 session_before_compact 钩子。这意味着所有场景（代码生成、内容创作、对话）都使用同一套通用摘要逻辑。

十四、关键设计决策总结

✅ 设计优势
  • 增量摘要避免重复处理
  • Tool result 截断大幅减少摘要输入量
  • Split Turn 处理保证极端情况下不丢上下文
  • 累积式文件追踪跨会话保留
  • Extension 钩子允许完全自定义
  • 利用真实 usage 数据减少估算误差
⚠️ 局限
  • chars/4 估算不精确（对中文可能偏差更大）
  • tool result 截断可能丢失关键信息
  • 摘要质量完全依赖 LLM 能力
  • 无法选择性保留特定消息（按 token 预算一刀切）
  • compaction 本身也消耗 token（摘要请求）
  • 不支持"pinned messages"（固定不被摘要的消息）
